1. THE ROOT CAUSE — WHY THE NATIVE MODULE WAS NULL FOR 16 BUILDS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Expo's autolinking (`expo-modules-autolinking`) uses a function called
`listFilesInDirectories` to find .podspec files inside a local module.

This function scans SUBDIRECTORIES ONLY. It NEVER reads files at the root
of the module directory. So if the podspec is at:

    modules/photo-saver/photo-saver.podspec   ← ROOT LEVEL — NOT FOUND

...autolinking silently skips the module. The JS side loads fine but the native
module is never compiled, causing:

    Error: Cannot find native module 'PhotoSaver'

THE FIX: Add "podspecPath" to expo-module.config.json under the "ios" key.
This bypasses the directory scan and tells autolinking exactly where the
podspec is:

    expo-module.config.json:
    {
      "platforms": ["ios"],
      "ios": {
        "modules": ["PhotoSaverModule"],
        "podspecPath": "./photo-saver.podspec"   ← THIS IS THE FIX
      }
    }

Without podspecPath → module returns null → not linked → native code never runs.
With podspecPath → autolinking uses it directly → Swift compiled → module works.

2. TWO THINGS BOTH REQUIRED — JS SIDE AND NATIVE SIDE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

There are TWO separate systems that must both be configured:

  A) Metro bundler (JS resolution) — needs a "file:" entry in root package.json:
       "photo-saver": "file:./modules/photo-saver"
     Without this: require('photo-saver') throws "Cannot find module"

  B) CocoaPods (native compilation) — needs the podspec + podspecPath:
       modules/photo-saver/photo-saver.podspec
       + "podspecPath" in expo-module.config.json
     Without this: JS loads but native module is null

Missing A → JS error ("Cannot find module 'photo-saver'")
Missing B → Native error ("Cannot find native module 'PhotoSaver'")
Both required for the module to work end to end.

4. WHY PHAssetCreationRequest.creationDate DOES NOT WORK
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

You might try:
    creationRequest.creationDate = dateTaken

iOS Photos COMPLETELY IGNORES this for JPEG files. It reads the date from
the EXIF metadata embedded in the file itself, not from the creation request.

THE ONLY RELIABLE FIX: Embed DateTimeOriginal into the JPEG EXIF before
saving using CGImageDestination. See the Swift code in Section 6.

The embedExifDate() function:
  1. Reads the image file into a CGImageSource
  2. Copies existing EXIF properties
  3. Overwrites DateTimeOriginal, DateTimeDigitized (EXIF) and DateTime (TIFF)
  4. Writes a new JPEG to a temp file via CGImageDestination
  5. Saves the new temp file (with correct EXIF) to Photos
  6. Deletes the temp file after saving

5. iOS PHOTOS PERMISSION LEVELS AND ALBUM CREATION
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

iOS has two Photos permission levels:

  FULL ACCESS (.authorized):
    - Can create a named album
    - Can add photo to a specific album
    - Photo appears in "Albums → [Event Name]"

  LIMITED ACCESS (.limited):
    - CANNOT create or modify albums
    - Photo saves to main library only (Camera Roll)
    - Must handle this case gracefully — don't reject, resolve with "limited_access"

Always request .readWrite permission, not just .addOnly.
Check status before attempting album operations.

In Swift:
    PHPhotoLibrary.requestAuthorization(for: .readWrite) { status in
        let hasFullAccess = (status == .authorized)
        // use hasFullAccess to decide whether to attempt album creation
    }

import ExpoModulesCore
import Photos
import ImageIO

public class PhotoSaverModule: Module {
  public func definition() -> ModuleDefinition {
    Name("PhotoSaver")

    AsyncFunction("saveToPhotos") { (fileUri: String, dateTakenMs: Double, albumName: String, promise: Promise) in
      guard let url = URL(string: fileUri) else {
        promise.reject("ERR_INVALID_URI", "Invalid file URI: \(fileUri)")
        return
      }

      let dateTaken: Date? = dateTakenMs > 0 ? Date(timeIntervalSince1970: dateTakenMs / 1000.0) : nil

      PHPhotoLibrary.requestAuthorization(for: .readWrite) { status in
        guard status == .authorized || status == .limited else {
          promise.reject("ERR_NO_PERMISSION", "Photos permission not granted")
          return
        }

        let hasFullAccess = (status == .authorized)

        // Embed correct EXIF date into file before saving (creationDate is ignored by iOS Photos)
        let saveUrl: URL
        if let dateTaken = dateTaken, let exifUrl = self.embedExifDate(sourceUrl: url, dateTaken: dateTaken) {
          saveUrl = exifUrl
        } else {
          saveUrl = url
        }

        let cleanup = { try? FileManager.default.removeItem(at: saveUrl) }

        if hasFullAccess {
          self.findOrCreateAlbum(named: albumName) { collection in
            guard let collection = collection else {
              cleanup()
              promise.reject("ERR_ALBUM_FAILED", "Could not find or create album '\(albumName)'")
              return
            }
            PHPhotoLibrary.shared().performChanges({
              if let creationRequest = PHAssetCreationRequest.creationRequestForAssetFromImage(atFileURL: saveUrl) {
                if let date = dateTaken { creationRequest.creationDate = date }
                if let placeholder = creationRequest.placeholderForCreatedAsset {
                  PHAssetCollectionChangeRequest(for: collection)?.addAssets([placeholder] as NSArray)
                }
              }
            }) { success, error in
              cleanup()
              if success {
                promise.resolve("ok")
              } else {
                promise.reject("ERR_SAVE_FAILED", error?.localizedDescription ?? "Failed to save photo")
              }
            }
          }
        } else {
          // Limited access — save to main library only, no album
          PHPhotoLibrary.shared().performChanges({
            if let creationRequest = PHAssetCreationRequest.creationRequestForAssetFromImage(atFileURL: saveUrl) {
              if let date = dateTaken { creationRequest.creationDate = date }
            }
          }) { success, error in
            cleanup()
            if success {
              promise.resolve("limited_access")
            } else {
              promise.reject("ERR_SAVE_FAILED", error?.localizedDescription ?? "Failed to save photo")
            }
          }
        }
      }
    }
  }

  // Embeds DateTimeOriginal into JPEG EXIF — the ONLY way iOS Photos respects the date
  private func embedExifDate(sourceUrl: URL, dateTaken: Date) -> URL? {
    guard let imageData = try? Data(contentsOf: sourceUrl),
          let source = CGImageSourceCreateWithData(imageData as CFData, nil),
          let uti = CGImageSourceGetType(source) else { return nil }

    let formatter = DateFormatter()
    formatter.dateFormat = "yyyy:MM:dd HH:mm:ss"
    let dateString = formatter.string(from: dateTaken)

    var properties = (CGImageSourceCopyPropertiesAtIndex(source, 0, nil) as? [String: Any]) ?? [:]

    var exifDict = (properties[kCGImagePropertyExifDictionary as String] as? [String: Any]) ?? [:]
    exifDict[kCGImagePropertyExifDateTimeOriginal as String] = dateString
    exifDict[kCGImagePropertyExifDateTimeDigitized as String] = dateString
    properties[kCGImagePropertyExifDictionary as String] = exifDict

    var tiffDict = (properties[kCGImagePropertyTIFFDictionary as String] as? [String: Any]) ?? [:]
    tiffDict[kCGImagePropertyTIFFDateTime as String] = dateString
    if tiffDict[kCGImagePropertyTIFFMake as String] == nil {
      tiffDict[kCGImagePropertyTIFFMake as String] = "Apple"
    }
    if tiffDict[kCGImagePropertyTIFFModel as String] == nil {
      tiffDict[kCGImagePropertyTIFFModel as String] = "iPhone"
    }
    properties[kCGImagePropertyTIFFDictionary as String] = tiffDict

    let tempUrl = URL(fileURLWithPath: NSTemporaryDirectory()).appendingPathComponent(UUID().uuidString + ".jpg")
    guard let destination = CGImageDestinationCreateWithURL(tempUrl as CFURL, uti, 1, nil) else { return nil }
    CGImageDestinationAddImageFromSource(destination, source, 0, properties as CFDictionary)
    guard CGImageDestinationFinalize(destination) else { return nil }

    return tempUrl
  }

  // Finds an existing album by name or creates a new one
  private func findOrCreateAlbum(named title: String, completion: @escaping (PHAssetCollection?) -> Void) {
    let fetchOptions = PHFetchOptions()
    fetchOptions.predicate = NSPredicate(format: "title = %@", title)
    let existing = PHAssetCollection.fetchAssetCollections(with: .album, subtype: .any, options: fetchOptions)

    if let collection = existing.firstObject {
      completion(collection)
      return
    }

    var placeholderIdentifier: String?
    PHPhotoLibrary.shared().performChanges({
      let request = PHAssetCollectionChangeRequest.creationRequestForAssetCollection(withTitle: title)
      placeholderIdentifier = request.placeholderForCreatedAssetCollection.localIdentifier
    }) { success, _ in
      guard success, let identifier = placeholderIdentifier else {
        completion(nil)
        return
      }
      let created = PHAssetCollection.fetchAssetCollections(withLocalIdentifiers: [identifier], options: nil)
      completion(created.firstObject)
    }
  }
}

7. JPEG vs JPEG — FILENAME NORMALISATION
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

When photos are uploaded, normalise .jpeg extension to .jpg before storing
the filename in the database. Both are the same format — .jpg is the
universally preferred extension.

Example (TypeScript, at upload time):
    const normalisedName = filename.replace(/\.jpeg$/i, '.jpg');

NOTE: iOS Photos info panel always displays "JPEG" as the format label
regardless of whether the file extension is .jpg or .jpeg. This is Apple's
display name for the format type — it is NOT the file extension and cannot
be changed. There is nothing wrong if you see "JPEG" there.

8. HEIC/HEIF PHOTOS SAVED TO "OTHER" INSTEAD OF PHOTOS LIBRARY
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

When a HEIC/HEIF file contains Live Photo metadata (a reference to a paired
.mov video file) but the .mov is absent, iOS classifies the image as an
INCOMPLETE LIVE PHOTO and places it in the "Other" category in Photos instead
of the main library. The user cannot see it in their normal photo grid.

This happens when:
  - A user uploads a Live Photo from their iPhone (HEIC + .mov pair)
  - Only the HEIC is stored/downloaded (the .mov is not included)
  - The HEIC retains the Live Photo UUID metadata pointing to a missing .mov

FIX OPTIONS:
  A) Convert HEIC to JPEG at upload time (recommended — MomentsInFrame approach)
     — strips all Live Photo metadata automatically
     — consistent format across all devices
     — no special handling needed at download time

  B) Strip Live Photo metadata before saving:
     Remove kCGImagePropertyMakerAppleDictionary key 17 (Live Photo identifier)
     from the image properties before writing via CGImageDestination

  C) Also save the paired .mov alongside the HEIC (complex, large files)

MomentsInFrame converts all uploads to JPEG on the server side, so this
issue does not occur at download time.

10. UPLOAD LOGIC — BOTH PLATFORMS (2-PHASE FLOW)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Upload is split into 2 phases to bypass Vercel's 4.5MB serverless body limit.

PHASE 1 — Get presigned URL (/api/upload-url):
  - App sends filename + event slug to server
  - Server generates a presigned S3 URL pointing to a staging/ prefix
  - App receives the URL + stagingKey

PHASE 2 — App PUTs file directly to S3 (bypasses Vercel entirely):
  - App uploads the raw file bytes straight to S3 using the presigned URL
  - No file bytes go through the Next.js function — only small JSON payloads

PHASE 3 — Process (/api/upload):
  - App sends { eventSlug, stagingKey, originalFilename, uploaderMobile, uploaderName }
  - Server fetches the staged file from S3 (server-to-S3 has no size limit)
  - Server runs: EXIF extraction → HEIC pre-decode → JPEG conversion → pHash → duplicate check → store
  - Staging file deleted from S3 after processing (success or failure)

FILENAME NORMALISATION AT UPLOAD:
  - All photos stored with .jpg extension regardless of original format
  - originalFilename stored in DB as .jpg: `filename.replace(/\.[^.]+$/, ".jpg")`
  - original_format column stores the original extension (heic, png, jpeg, etc.)

PRE-UPLOAD DEDUPLICATION (client side — before API call):
  - App fetches all existing filenames for the event before starting
  - Any asset whose filename already exists is skipped immediately
  - Prevents re-uploading the same file in the same session without any API calls

WHAT GETS STORED IN S3 PER PHOTO (3 versions):
  - Full original: stored as-is (JPEG) at event_id/uuid.jpg
  - Thumbnail: 300px wide, JPEG quality 75, for grid view
  - Display: 1200px wide, JPEG quality 85, for full-screen view

HEIC HANDLING:
  - iPhone HEIC files (Live Photos, burst) have 48+ iref box references
  - sharp/libvips has a 16-reference security limit → crashes on these files
  - Fix: pre-decode HEIC using heic-convert (WASM, no limit) BEFORE passing to sharp
  - HEIC is converted to JPEG before duplicate check so file sizes are comparable
  - Without this, every re-upload of a HEIC looks like a quality upgrade (wrong size)

VIDEO UPLOADS:
  - Currently blocked server-side ("Video uploads not available at this time")
  - Videos pass through unconverted in the trial build — pHash is image-only

--- CODE: /api/upload-url/route.ts (Phase 1 — get presigned URL) ---
// App calls this first. Returns a presigned S3 URL + stagingKey.
// App then PUTs the file directly to S3 — never through Vercel.
export async function POST(request: NextRequest) {
  const { eventSlug, filename, contentType } = await request.json();

  // Validate event exists and is not expired
  const { data: event } = await supabaseAdmin
    .from("events")
    .select("id, expires_at")
    .eq("slug", eventSlug)
    .single();

  const extension = filename.split(".").pop()?.toLowerCase() ?? "";
  const stagingKey = `staging/${event.id}/${randomUUID()}.${extension}`;

  const uploadUrl = await getPresignedUploadUrl(
    stagingKey,
    contentType || "application/octet-stream"
  );

  return NextResponse.json({ uploadUrl, stagingKey });
}

export async function POST(request: NextRequest) {

  // Normalise filename: .heic/.jpeg/.png → .jpg
  const originalExtension = getExtension(originalFilename);
  const isVideo = VIDEO_EXTENSIONS.has(originalExtension);
  const storedFilename = isVideo
    ? originalFilename
    : originalFilename.replace(/\.[^.]+$/, ".jpg");

  // Fetch staged file from S3 (no size limit server-to-S3)
  const getResult = await s3Client.send(
    new GetObjectCommand({ Bucket: S3_BUCKET, Key: stagingKey })
  );
  let inputBuffer = await streamToBuffer(getResult.Body as NodeJS.ReadableStream);

  // 1. Extract EXIF date BEFORE any conversion
  const { takenAt, dateSource } = await extractBestDate(tempInputPath);

  // 2a. Pre-decode HEIC (iPhone HEICs have 48+ iref refs; sharp limit is 16)
  const decodedBuffer = await preDecodeIfHeic(inputBuffer, originalExtension);
  const isHeic = ["heic", "heif"].includes(originalExtension);

  // 2b. Convert HEIC to JPEG NOW (before duplicate check) so sizes are comparable
  const convertedBuffer = isHeic ? await convertToJpg(decodedBuffer) : decodedBuffer;
  const newPhash = await computePhash(convertedBuffer);

  // 3. Duplicate detection against all existing photos in this event
  const { data: existingPhotos } = await supabaseAdmin
    .from("photos")
    .select("id, phash, file_size_bytes, taken_at, date_source")
    .eq("event_id", event.id)
    .eq("media_type", "photo")
    .is("deleted_at", null)
    .not("phash", "is", null);

  for (const existing of existingPhotos ?? []) {
    if (!isDuplicate(existing.phash, newPhash, existing.file_size_bytes, uploadBuffer.length,
        existing.taken_at ? new Date(existing.taken_at) : null, takenAt,
        existing.date_source, dateSource)) continue;

    const resolution = resolveDuplicate({ ...existing }, { phash: newPhash, fileSizeBytes: uploadBuffer.length, takenAt, dateSource });

    if (resolution.action === "merge_self_duplicate" || resolution.action === "keep_existing") {
      return NextResponse.json({ success: true, duplicate: true, ... });
    }

    if (resolution.action === "keep_new") {
      // Generate new thumbnail + display, upload 3 versions to S3, UPDATE existing DB row in-place
      await supabaseAdmin.from("photos").update({ storage_path, thumbnail_path, display_path, phash: newPhash, ... }).eq("id", existing.id);
      return NextResponse.json({ success: true, upgraded: true, ... });
    }
  }

  const [thumbBuffer, displayBuffer] = await Promise.all([
    sharp(uploadBuffer).rotate().resize(300).jpeg({ quality: 75 }).toBuffer(),
    sharp(uploadBuffer).rotate().resize(1200).jpeg({ quality: 85 }).toBuffer(),
  ]);
  await supabaseAdmin.from("photos").insert({ event_id, storage_path, thumbnail_path, display_path, phash: newPhash, taken_at, date_source, file_size_bytes, original_filename: storedFilename, original_format: originalExtension, ... });

  // Cleanup: delete temp file + staging S3 key (always, success or failure)
  // ...in finally block
}

Android uses a custom local native module (modules/media-store) that wraps
Android's MediaStore API. iOS uses PhotoSaver (see sections 1–9).

SINGLE PHOTO DOWNLOAD:
  1. Download photo from server to Expo cache directory (temp file)
  2. Pass local path to MediaStore.saveToDownloads(path, filename, folderName, mimeType, dateTakenMs)
  3. MediaStore saves it to Downloads/[EventFolderName]/filename.jpg
  4. dateTakenMs sets the correct "date taken" in Android's media database
  5. Delete temp cache file after saving

ZIP DOWNLOAD (bulk):
  - Same flow but mimeType is 'application/zip' and no dateTakenMs needed
  - File saved to Downloads/[EventFolderName]/slug-photos.zip
  - For >25 photos, split into batches: slug-photos-part1of3.zip etc.

FOLDER NAMING:
  - On first download, user is asked to name the folder (defaults to event name)
  - Folder name stored in SecureStore keyed by event slug
  - Not asked again for the same event — reuses saved folder name
  - Folder created if it doesn't exist before saving

KEY DIFFERENCE FROM iOS:
  - Android reads date from dateTakenMs passed to MediaStore — no EXIF embedding needed
  - iOS completely ignores creation date passed in code — must embed EXIF into file
  - Android saves to Downloads folder (file manager accessible)
  - iOS saves to Photos app (camera roll / named album)

iOS ZIP:
  - iOS has no Downloads folder accessible to apps
  - ZIP files are shared via iOS Share Sheet (Sharing.shareAsync) instead
  - User saves to Files app, AirDrop, etc. — their choice

    const cacheUri = `${FileSystem.cacheDirectory}${filename}`;
    const dlResult = await FileSystem.downloadAsync(url, cacheUri);
    if (dlResult.status !== 200) throw new Error(`HTTP ${dlResult.status}`);
    if (PhotoSaver) {
      const result = await PhotoSaver.saveToPhotos(cacheUri, dateTakenMs ?? 0, params.name);
      if (result === 'limited_access') {
        // Only show this alert once — user has Limited Photos permission
        const alreadyShown = await SecureStore.getItemAsync('mif_limited_access_shown');
        if (!alreadyShown) {
          await SecureStore.setItemAsync('mif_limited_access_shown', '1');
          Alert.alert('Photo Saved',
            'To organise photos into your event album, allow Full Access in Settings → Privacy & Security → Photos.',
            [{ text: 'OK' }]);
        }
      }
    } else {
      await MediaLibrary.saveToLibraryAsync(cacheUri); // fallback if PhotoSaver unavailable
    }
    await FileSystem.deleteAsync(cacheUri, { idempotent: true });
  }
}

12. DUPLICATE DETECTION LOGIC (server-side, /api/upload)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Uses perceptual hashing (pHash) via sharp-phash. Detects visually identical
photos even when one has been recompressed (e.g. WhatsApp copies).

THRESHOLDS:
  - HASH_THRESHOLD = 18 bits — outer limit, beyond this never a duplicate
  - HASH_THRESHOLD_STRICT = 8 bits — tight match, almost certainly same photo
  - SIZE_SIMILARITY_RATIO = 15% — files within 15% size are "same quality"

isDuplicate() RULES (checked in order):
  1. If hash distance > 18 → NOT a duplicate (different photos)
  2. If distance 9–18 (loose band):
     - Match ONLY if exactly one side has EXIF (WhatsApp copy scenario)
     - OR both have EXIF + timestamps within 2s + sizes differ >15% (WhatsApp HD)
     - Otherwise → NOT a duplicate (burst shots look similar but aren't the same)
  3. If both have EXIF and timestamps differ by >2s → NOT a duplicate
     - Exception: if distance ≤8 AND sizes differ >15% → WhatsApp shifted the
       timestamp to receive time; trust the visual hash over the timestamp
  4. If one side has EXIF and size differs >90% AND distance >8 → NOT a duplicate
     (hash collision guard — 40-70% is normal for WhatsApp, 90%+ is suspicious)
  5. Otherwise → IS a duplicate

resolveDuplicate() — what to do when a duplicate IS found:
  - Self-duplicate (same quality, same date source, same timestamp): merge silently,
    no error shown to user. Prevents double-upload of same original.
  - New upload has larger file OR has EXIF and existing doesn't → keep_new (UPGRADE)
  - Existing is equal or better quality → keep_existing (SKIP)

RESULT STATUSES returned to app:
  - 'success'   — new photo added
  - 'duplicate' — existing photo kept, new upload skipped
  - 'upgraded'  — existing photo replaced with higher-quality version
  - 'failed'    — error during processing
  - 'cancelled' — user cancelled mid-upload

IMPORTANT: Duplicate detection is photos-only. Videos are not deduplicated.
pHash is computed AFTER HEIC→JPEG conversion so sizes are comparable.

--- CODE: src/lib/duplicate-detection.ts ---

import phash from "sharp-phash";

const HASH_THRESHOLD = 18;        // outer limit — beyond this, different photos
const HASH_THRESHOLD_STRICT = 8;  // tight match — almost certainly same photo
const SIZE_SIMILARITY_RATIO = 0.15; // within 15% size = same quality copy

// Hamming distance: number of differing bits between two hex pHash strings
function hammingDistance(hashA: string, hashB: string): number {
  const a = BigInt("0x" + hashA);
  const b = BigInt("0x" + hashB);
  let xor = a ^ b;
  let distance = 0;
  while (xor > 0n) {
    distance += Number(xor & 1n);
    xor >>= 1n;
  }
  return distance;
}

export function isDuplicate(
  hashA: string, hashB: string,
  sizeA: number, sizeB: number,
  takenAtA?: Date | null, takenAtB?: Date | null,
  dateSourceA?: string | null, dateSourceB?: string | null,
): boolean {
  const distance = hammingDistance(hashA, hashB);
  if (distance > HASH_THRESHOLD) return false; // rule 1: clearly different

  const oneHasExif = (dateSourceA === "exif") !== (dateSourceB === "exif");
  const bothHaveExif = dateSourceA === "exif" && dateSourceB === "exif";
  const timestampsClose = takenAtA != null && takenAtB != null &&
    Math.abs(takenAtA.getTime() - takenAtB.getTime()) <= 2000;
  const sizeDiffRatio = Math.abs(sizeA - sizeB) / Math.max(sizeA, sizeB);
  const sizeDiffSignificant = sizeDiffRatio > SIZE_SIMILARITY_RATIO;

  // Rule 2: loose band (9–18 bits) — only match WhatsApp scenarios
  if (distance > HASH_THRESHOLD_STRICT) {
    const isWhatsappHd = bothHaveExif && timestampsClose && sizeDiffSignificant;
    if (!oneHasExif && !isWhatsappHd) return false;
  }

  // Rule 3: both have EXIF but timestamps differ >2s → different photos
  // Exception: tight hash + big size diff = WhatsApp changed the timestamp
  const whatsappTimestampShift = distance <= HASH_THRESHOLD_STRICT && sizeDiffSignificant;
  if (bothHaveExif && !timestampsClose && takenAtA != null && takenAtB != null && !whatsappTimestampShift) {
    return false;
  }

  // Rule 4: hash collision guard — >90% size diff + loose hash = false positive
  if (oneHasExif && sizeDiffRatio > 0.90 && distance > HASH_THRESHOLD_STRICT) return false;

  if (oneHasExif) return true;
  if (bothHaveExif && timestampsClose && sizeDiffSignificant) return true; // WhatsApp HD
  if (whatsappTimestampShift) return true;

  // Burst shot protection: same camera, within 2s, similar size, but visually different (distance > 0)
  if (bothHaveExif && timestampsClose && !sizeDiffSignificant && distance > 0) return false;

  return sizeDiffRatio <= SIZE_SIMILARITY_RATIO;
}

13. QUALITY UPGRADE LOGIC (part of duplicate detection)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

An upgrade happens when a duplicate IS detected but the NEW upload is better
than what's already stored. File size is used as the quality proxy.

WHEN AN UPGRADE HAPPENS:
  - New file is larger (less compressed) than existing → new wins
  - New file has genuine EXIF date and existing only has a fallback timestamp → new wins
  - Common real-world scenario: someone uploads a WhatsApp copy first (compressed,
    no EXIF), then the original owner uploads their original (larger, has EXIF)

WHAT HAPPENS ON UPGRADE (server-side):
  1. Convert new file to JPEG if not already
  2. Generate new thumbnail (300px) and display version (1200px)
  3. Upload all 3 versions to S3 (new paths, old paths abandoned)
  4. UPDATE the existing photo DB row in-place (same photo ID, same position in timeline)
  5. Update: storage_path, thumbnail_path, display_path, phash, taken_at, date_source,
     file_size_bytes, original_filename, original_format, uploader info

WHY UPDATE IN-PLACE (not insert):
  - Preserves the photo's position in the event timeline
  - Keeps the same photo ID (no broken references)
  - The WhatsApp copy "becomes" the original — no duplicates in the gallery

WHAT USER SEES IN APP:
  - Upload summary shows "X better quality photos uploaded"
  - Upgrade appears in the duplicate viewer alongside skipped duplicates
  - Message shown: "Replaced a lower quality WhatsApp copy already in Other Photos Gallery"
  - Notification panel shows upgradesFound count in upload history

QUALITY UPGRADE IS CRITICAL — DO NOT MODIFY without explicit permission.

--- CODE: src/lib/duplicate-detection.ts — resolveDuplicate() ---

export type DuplicateResolution =
  | { action: "keep_new"; reason: string }
  | { action: "keep_existing"; reason: string }
  | { action: "merge_self_duplicate"; reason: string };

export function resolveDuplicate(
  existing: PhotoCandidate,  // { id, phash, fileSizeBytes, takenAt, dateSource }
  incoming: NewUpload        // { phash, fileSizeBytes, takenAt, dateSource }
): DuplicateResolution {
  const sizeDiff = incoming.fileSizeBytes - existing.fileSizeBytes;
  const sameQuality = Math.abs(sizeDiff) < existing.fileSizeBytes * 0.02; // within 2% = same
  const sameDateSource = incoming.dateSource === existing.dateSource;
  const sameTimestamp = Math.abs(incoming.takenAt.getTime() - existing.takenAt.getTime()) < 1000;

  // Self-duplicate: same guest re-uploading their own original by mistake.
  // All three must match: quality, date source, timestamp. Merge silently.
  if (sameQuality && sameDateSource && sameTimestamp) {
    return { action: "merge_self_duplicate", reason: "Identical quality and timestamp — accidental repeat upload. Merged silently." };
  }

  const incomingHigherQuality = sizeDiff > 0 && !sameQuality;
  const incomingMoreTrustworthyDate = incoming.dateSource === "exif" && existing.dateSource !== "exif";

  // Upgrade: new upload is better quality OR has a more trustworthy date
  if (incomingHigherQuality || incomingMoreTrustworthyDate) {
    return {
      action: "keep_new",
      reason: incomingHigherQuality
        ? "New upload is higher quality (larger file, less compression)."
        : "New upload has genuine EXIF date; existing only has fallback timestamp.",
    };
  }

  // Skip: existing is equal or better
  return { action: "keep_existing", reason: "Existing photo is equal or higher quality." };
}
